    <script>
        const SCRIPT_URL = 'https://script.google.com/macros/s/AKfycbyctWwwjvLVjmRWfMBho9yHdCcsO2q_88ji8roNvLRb5EeA8dZKrxtWXQ8EiU2cBfmrCg/exec';
        let currentUserData = null;

        async function processLogin() {
            const emailInp = document.getElementById('email').value.toLowerCase().trim();
            const passInp = document.getElementById('pass').value.trim();
            const errMsg = document.getElementById('error-msg');
            errMsg.style.display = "block"; errMsg.innerText = "Verifying...";

            try {
                const res = await fetch(`${SCRIPT_URL}?sheet=Users`);
                const users = await res.json();
                const found = users.find(u => String(u.email).toLowerCase().trim() === emailInp && String(u.password).trim() === passInp);
                if (found) {
                    currentUserData = found;
                    document.getElementById('login-screen').style.display = 'none';
                    document.getElementById('dashboard').style.display = 'block';
                    document.getElementById('user-display').innerText = "Hi, " + found.name;
                    loadStudentData();
                } else { errMsg.innerText = "Invalid Login!"; }
            } catch (e) { errMsg.innerText = "Error!"; }
        }

        async function loadStudentData() {
            const myEmail = currentUserData.email.toLowerCase().trim();
            const [att, ex, ints, certs] = await Promise.all([
                fetch(`${SCRIPT_URL}?sheet=Attendance`).then(r => r.json()),
                fetch(`${SCRIPT_URL}?sheet=Exams`).then(r => r.json()),
                fetch(`${SCRIPT_URL}?sheet=Interviews`).then(r => r.json()),
                fetch(`${SCRIPT_URL}?sheet=Certificates`).then(r => r.json())
            ]);

            document.getElementById('attendance-rows').innerHTML = att.filter(r => String(r.email).toLowerCase().trim() === myEmail).map(r => 
                `<tr><td>${r.date}</td><td><b>${r.topic}</b></td><td><span class="badge ${String(r.status).toLowerCase()}">${r.status}</span></td></tr>`
            ).join('');

            document.getElementById('exam-rows').innerHTML = ex.filter(r => String(r.email).toLowerCase().trim() === myEmail).map(r => {
                let actionHtml = String(r.status).toLowerCase() === 'pending' 
                    ? `<a href="${r.exam_link || '#'}" target="_blank" class="start-btn">▶ Start Exam</a>` 
                    : `<span class="status-done">✅ Completed</span>`;
                return `<tr><td>${r.subject}</td><td>${r.topic}</td><td>${r.total_q || '-'}</td><td>${r.time || '-'}</td><td>${actionHtml}</td></tr>`;
            }).join('');

            document.getElementById('int-rows').innerHTML = ints.filter(r => String(r.email).toLowerCase().trim() === myEmail).map(r => {
                let cleanDate = String(r.date).includes('T') ? r.date.split('T')[0] : r.date;
                return `<tr><td>${cleanDate}</td><td><b>${r.type}</b></td><td>${r.comm}</td><td>${r.confidence}</td><td>${r.technical}</td><td><b>${r.overall}</b></td></tr>`;
            }).join('');

            const myCert = certs.find(r => String(r.email).toLowerCase().trim() === myEmail);
            if(myCert && (myCert.status === "Ready" || myCert.status === "Issued")) {
                document.getElementById('cert-action').innerHTML = `<h3>Congratulations!</h3><a href="${myCert.download_link}" target="_blank" class="btn-blue" style="padding:15px 30px; text-decoration:none;">Download Certificate</a>`;
            } else {
                document.getElementById('cert-action').innerHTML = `<p>Certificate will be available after completion.</p>`;
            }
        }

        function switchTab(id, btn) {
            document.querySelectorAll('.tab-content').forEach(c => c.style.display = 'none');
            document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
            document.getElementById(id).style.display = 'block';
            btn.classList.add('active');
        }
    </script>
